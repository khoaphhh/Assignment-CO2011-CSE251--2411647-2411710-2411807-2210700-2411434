# Tạo và kích hoạt môi trường ảo
    Tạo môi trường: python -m venv venv
    Kích hoạt:
        # On Windows
        venv\Scripts\activate
        # On Linux/macOS
        source venv/bin/activate

# Tải thư viện cần thiết 
    Tải thư viện: pip install -r requirements.txt
    Cập nhật thư viện: pip install --upgrade -r requirements.txt


# Chạy chương trình
    python src/main.py

    Đối với task 5, nhập hàm mục tiêu dạng 'p1=2 p3=-1 p4=5'
    Ví dụ trong deadlock_example.pnml: nhập 'p1_think=1 p2_think=2 fork1=3' (các trọng số khác = 1)
